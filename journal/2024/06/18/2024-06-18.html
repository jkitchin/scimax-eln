from scipy.integrate import solve_ivp
import nava

@SqlCache
def solution(k):
    '''Get solution for k.'''
    nu = 1
    Vspan = (0, 2)
    Fa0 = 1

    # define the ODE
    def ode(V, Fa, k):
        return -k * Fa / nu

    sol = solve_ivp(ode, Vspan, (Fa0,), args=(k,))
    nava.play('../../../../cash-register-fake-88639.mp3')
    return sol








K = [0.1, 0.2, 0.4]
