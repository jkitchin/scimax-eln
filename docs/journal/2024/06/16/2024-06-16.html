from pycse.hashcache import SqlCache

def serialize_atoms(self, obj):
    if isinstance(obj, ase.atoms.Atoms):
        return obj.todict()
    else:
        raise TypeError
    
SqlCache.default = serialize_atoms

import ase
from ase.build import bulk

atoms = bulk('Cu')
